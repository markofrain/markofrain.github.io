addRow("#treeTableList", tpl, data, rootId, true);

//list:tbody标签id，tpl为js中解析的模板,data为将areaList解析的json字符串，pid为父id，root表示根节点值
function addRow(list, tpl, data, pid, root){
    for (var i=0; i<data.length; i++){
        var row = data[i];
        if ((${fns:jsGetVal('row.parentId')}) == pid){
            //使用js Mustache模板库，render回调设置模板参数和值
            $(list).append(Mustache.render(tpl, {
                dict: {
                    type: getDictLabel(${fns:toJson(fns:getDictList('sys_area_type'))}, row.type)
                }, 
                pid: (root?0:pid),
                row: row
            }));
            addRow(list, tpl, data, row.id);
        }
    }
}

菜单管理

List: 查询所有菜单，并创建一个新List

通过Menu类的静态sort方法惊醒排序，将sourcelist排好序放在新创建的list中

Menu.sortList(list, sourcelist, Menu.getRootId(), true);

public static void sortList(List<Menu> list, List<Menu> sourcelist, String parentId, boolean cascade){
    for (int i=0; i<sourcelist.size(); i++){
        Menu e = sourcelist.get(i);
        if (e.getParent()!=null && e.getParent().getId()!=null
                && e.getParent().getId().equals(parentId)){
            list.add(e);
            if (cascade){
                // 判断是否还有子节点, 有则继续获取子节点
                for (int j=0; j<sourcelist.size(); j++){
                    Menu child = sourcelist.get(j);
                    if (child.getParent()!=null && child.getParent().getId()!=null
                            && child.getParent().getId().equals(e.getId())){
                        sortList(list, sourcelist, e.getId(), true);
                        break;
                    }
                }
            }
        }
    }
}

该方法简而言之，从sourcelist中一次获得对象，判断该菜单的父id是否为参数的parentId，则说明是其子类，并添加到list，随后判断该节点是否有子节点，再次循环sourcelise，如果某个菜单id的父id等于刚才被添加到list中的菜单，那么是父子关系则add到list，因为多级菜单，所以用到迭代方法，，只不过第一次调用此方法是，传递parentid是根root。 该代码与区域管理都是tree table，排序的方式也是一样的。不过区域排序是在前台js写的。 form: 还是和区域管理一样，一个按钮，一个超链接创建下级，创建实例时，一个parent为root，一个为当前id。 为传入form页面的新menu对象设置sort值，方式：排序，值通过最大值+30 Save: 上级菜单选项，使用tree select树形选择框，请求路径/sys/menu/treeData。显示树形选择 保存时,验证是否为管理员权限，之后进行服务端参数有效性校验 Delete: 删除,逻辑删除，更新del_flag字段为1 menuTree页面是最左侧导航 menuTreeSelect
